# creating virtual environment:
user@host ~/src/daily-tasks $ python -m venv daily-tasks-env

# activating environment:
user@host ~/src/daily-tasks $ source ./daily-tasks-env/Scripts/activate

# deactivating environment:
user@host ~/src/daily-tasks $ deactivate

# installing environment:
user@host ~/src/daily-tasks $ python -m pip install -r requirements.txt

# storing environment requirements:
user@host ~/src/daily-tasks $ python -m pip freeze > requirements.txt

# executing tasks:
user@host ~/src/daily-tasks $ ./tasks.py
